CAMBIOS EN Program.cs (DESPUÉS) ✅
───────────────────────────────────────────────────────────────────────────
using CapaDatos.Interfaces;
using CapaDatos.Repositorio;
using CapaServicio.Interfaces;
using CapaServicio.Servicios;

var builder = WebApplication.CreateBuilder(args);

// Obtener connection string
var connectionString = builder.Configuration.GetConnectionString("DefaultConnection") 
    ?? throw new InvalidOperationException("No se encontró la cadena de conexión");

// Registrar repositorios y servicios
builder.Services.AddScoped<ITransportistaRepository>(sp => 
    new TransportistaRepository(connectionString));
builder.Services.AddScoped<ITransportistaService, TransportistaService>();

// CORS
builder.Services.AddCors(options =>
{
    options.AddPolicy("AllowAll", policy =>
    {
        policy.AllowAnyOrigin()
              .AllowAnyMethod()
              .AllowAnyHeader();
    });
});

builder.Services.AddControllers();
builder.Services.AddOpenApi();

var app = builder.Build();

if (app.Environment.IsDevelopment())
{
    app.MapOpenApi();
}

app.UseCors("AllowAll");
app.UseHttpsRedirection();
app.UseAuthorization();
app.MapControllers();
app.Run();

PRÓXIMOS PASOS SUGERIDOS
═════════════════════════════════════════════════════════════════════════════

1. Actualizar connection string en appsettings.json con tus datos reales

2. Ejecutar ScriptTransportistas.sql en SQL Server para crear la tabla

3. Ejecutar: dotnet build

4. Ejecutar: dotnet run --project API_TRANSPORTISTE

5. Acceder a: https://localhost:5001/api/transportista

6. Probar con Postman/Swagger
